#include <tunables/global>

profile reminder_app flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>
  #include <abstractions/nameservice>
  #include <abstractions/user-tmp>

  # Specific capabilities only
  capability net_bind_service,
  capability setgid,
  capability setuid,
  
  # Network access
  network inet stream,
  network inet6 stream,
  network inet dgram,
  network inet6 dgram,
  
  # Signals
  signal (send) set=(kill,term,int,hup,cont),

  # Executable access
  /run.sh rix,
  /usr/bin/node rix,
  /usr/bin/pnpm rix,
  /usr/bin/pg_isready rix,
  /bin/bash rix,
  /bin/sh rix,
  /usr/bin/env rix,
  
  # Data and config
  /data/** rw,
  
  # Application files
  /app/** r,
  /app/.next/** rw,
  /app/node_modules/** r,
  /app/public/** r,
  
  # Temporary files
  /tmp/** rw,
  
  # System libraries (read-only)
  /etc/ld.so.cache r,
  /usr/lib/** rm,
  /lib/** rm,
  
  # Configuration files (read-only)
  /etc/hosts r,
  /etc/resolv.conf r,
  /etc/nsswitch.conf r,
  /etc/ssl/openssl.cnf r,
  /etc/ssl/certs/** r,
  
  # Logging support
  /var/log/** w,
  /dev/log w,
  /dev/null rw,
  /dev/tty rw,
  /dev/urandom r,
  
  # Process info (read-only)
  /proc/*/stat r,
  /proc/*/status r,
  /proc/cpuinfo r,
  /proc/meminfo r,

  # Deny dangerous access
  deny /sys/firmware/** rwklx,
  deny /proc/sys/kernel/** w,
  deny /etc/passwd w,
  deny /etc/shadow rwklx,
  deny /root/** rwklx,
}
